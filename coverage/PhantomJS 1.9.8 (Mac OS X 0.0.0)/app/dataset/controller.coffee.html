"use strict";
var app;
 
app = angular.module("app.controllers");
 
app.controller("DatasetCtrl", [
  "$scope", "$rootScope", "$log", "DataFactory", "UserFactory", "TableService", "MapService", "ConverterService", "$timeout", "ProgressService", "$stateParams", "$location", "$translate", "ngToast", "DataService", "VisualizationService", function($scope, $rootScope, $log, DataFactory, UserFactory, Table, Map, Converter, $timeout, Progress, $stateParams, $location, $translate, ngToast, Data, Visualization) {
    var datasetId;
    $rootScope.link = $location.$$absUrl;
    $rootScope.showVidatioLink = false;
    datasetId = $stateParams.id;
    $scope.information = [];
    DataFactory.get({
      id: datasetId
    }, function(data) {
      var category, created, dataOrigin, description, image, parent, tags, title, updated, userName;
      $scope.data = data;
      updated = new Date($scope.data.updatedAt);
      created = new Date($scope.data.createdAt);
      if ($scope.data.metaData != null) {
        Data.meta["fileType"] = $scope.data.metaData.fileType || "-";
      } else {
        Data.meta["fileType"] = "-";
      }
      tags = $scope.data.tags || "-";
      category = $scope.data.category || "-";
      dataOrigin = "Vidatio";
      userName = $scope.data.userId.name || "-";
      title = $scope.data.name || "Vidatio";
      parent = $scope.data.parentId || "-";
      image = $scope.data.image || "images/logo-greyscale.svg";
      description = $scope.data.description || "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Donec a diam lectus. Sed sit amet ipsum mauris. Maecenas congue ligula ac quam viverra nec consectetur ante hendrerit. Donec et mollis dolor. Praesent et diam eget libero egestas mattis sit amet vitae augue. Nam tincidunt congue enim, ut porta lorem lacinia consectetur.";
      return $scope.information.push({
        title: title,
        image: image,
        id: datasetId,
        created: created,
        creator: userName,
        origin: dataOrigin,
        updated: updated,
        description: description,
        parent: parent,
        category: category,
        tags: tags
      });
    }, function(error) {
      $log.info("DatasetCtrl error on get dataset from id");
      $log.error(error);
      return $translate("TOAST_MESSAGES.DATASET_COULD_NOT_BE_LOADED").then(function(translation) {
        return ngToast.create({
          content: translation,
          className: "danger"
        });
      });
    });
    $scope.createVidatio = function() {
      $log.info("DatasetCtrl $scope.createVidatio called");
      return $translate("OVERLAY_MESSAGES.READING_FILE").then(function(message) {
        Progress.setMessage(message);
        Data.createVidatio($scope.data);
        return $timeout(function() {
          return Progress.setMessage("");
        });
      });
    };
    $scope.downloadDataset = function() {
      $log.info("DatasetCtrl downloadDataset called");
      return this.createVidatio();
    };
    $scope.downloadImage = function() {
      $log.info("DatasetCtrl downloadImage called");
      return this.createVidatio();
    };
    $scope.toggleVidatioLink = function() {
      $log.info("DatasetCtrl getVidatioLink called");
      $log.debug({
        id: datasetId,
        link: $rootScope.link
      });
      return $rootScope.showVidatioLink = $rootScope.showVidatioLink ? false : true;
    };
    $scope.hideVidatioLink = function() {
      return $rootScope.showVidatioLink = false;
    };
    return $scope.copyVidatioLink = function() {
      var error, link, range, successful;
      $log.info("DatasetCtrl copyVidatioLink called");
      window.getSelection().removeAllRanges();
      link = document.querySelector("#vidatio-link");
      range = document.createRange();
      range.selectNode(link);
      window.getSelection().addRange(range);
      try {
        successful = document.execCommand("copy");
        $log.debug({
          message: "DatasetCtrl copy vidatio-link to clipboard",
          successful: successful
        });
        $translate("TOAST_MESSAGES.LINK_COPIED").then(function(translation) {
          return ngToast.create({
            content: translation
          });
        });
      } catch (_error) {
        error = _error;
        $log.info("DatasetCtrl vidatio-link could not be copied to clipboard");
        $log.error({
          error: error
        });
        $translate("TOAST_MESSAGES.LINK_NOT_COPIED").then(function(translation) {
          return ngToast.create({
            content: translation,
            className: "danger"
          });
        });
      }
      return window.getSelection().removeAllRanges();
    };
  }
]);
