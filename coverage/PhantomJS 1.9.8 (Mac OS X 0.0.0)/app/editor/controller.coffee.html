"use strict";
var app;
 
app = angular.module("app.controllers");
 
app.controller("EditorCtrl", [
  "$scope", "$rootScope", "$log", "$timeout", "DataService", "ngToast", "$translate", function($scope, $rootScope, $log, $timeout, Data, ngToast, $translate) {
    var isFirefox, setTitleInputWidth, viewsToDisplay;
    $scope.editor = Data;
    isFirefox = navigator.userAgent.toLowerCase().indexOf('firefox') > -1;
    $scope.activeViews = 2;
    $scope.activeTabs = [false, true, false];
    viewsToDisplay = [true, true];
    $rootScope.showTableView = viewsToDisplay[0], $rootScope.showVisualizationView = viewsToDisplay[1];
    $translate("NEW_VIDATIO").then(function(translation) {
      $scope.standardTitle = translation;
      if ($scope.editor.name === "") {
        return $scope.editor.name = $scope.standardTitle;
      }
    });
    $timeout(function() {
      return $("#vidatio-title").css("width", setTitleInputWidth());
    });
    $scope.tabClicked = function(tabIndex) {
      var i, tab, _i, _len, _results;
      $log.info("EditorCtrl tabClicked called");
      for (i in $scope.activeTabs) {
        $scope.activeTabs[i] = false;
      }
      $scope.activeTabs[tabIndex] = true;
      if (tabIndex === 0) {
        viewsToDisplay = [true, false];
      } else if (tabIndex === 1) {
        viewsToDisplay = [true, true];
      } else {
        viewsToDisplay = [false, true];
      }
      $log.debug({
        message: "EditorCtrl tabClicked called",
        tabIndex: tabIndex,
        viewsToDisplay: viewsToDisplay
      });
      $rootScope.showTableView = viewsToDisplay[0], $rootScope.showVisualizationView = viewsToDisplay[1];
      $scope.activeViews = 0;
      _results = [];
      for (_i = 0, _len = viewsToDisplay.length; _i < _len; _i++) {
        tab = viewsToDisplay[_i];
        if (tab) {
          _results.push($scope.activeViews++);
        } else {
          _results.push(void 0);
        }
      }
      return _results;
    };
    $("#vidatio-title").on('input', function() {
      return $("#vidatio-title").css("width", setTitleInputWidth());
    });
    $scope.saveVidatioTitle = function() {
      $log.info("EditorCtrl saveVidatioTitle called");
      if ($scope.editor.name === "") {
        $scope.editor.name = $scope.standardTitle;
      }
      $timeout(function() {
        return $("#vidatio-title").css("width", setTitleInputWidth());
      });
      $log.debug({
        title: $scope.editor.name
      });
      return true;
    };
    return setTitleInputWidth = function() {
      var inputWidth;
      inputWidth = $("#vidatio-title").textWidth();
      if (isFirefox) {
        if (inputWidth < 150) {
          inputWidth = inputWidth * 1.5 + "px";
        } else {
          inputWidth = inputWidth * 1.4 + "px";
        }
      } else {
        inputWidth = inputWidth + 10 + "px";
      }
      return inputWidth;
    };
  }
]);
