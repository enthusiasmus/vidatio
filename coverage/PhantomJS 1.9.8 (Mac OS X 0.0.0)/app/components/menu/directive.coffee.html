"use strict";
var app;
 
app = angular.module("app.directives");
 
app.directive("menu", [
  "$timeout", "UserService", "$rootScope", "$log", "$state", function($timeout, UserService, $rootScope, $log, $state) {
    return {
      restrict: "E",
      templateUrl: "components/menu/menu.html",
      replace: true,
      link: function($scope, $element) {
        $scope.logout = function() {
          var element, _i, _len, _ref, _results;
          $log.info("Menu directive logout called");
          UserService.logout();
          if (!$rootScope.history.length) {
            $log.info("UserCtrl redirect to app.index");
            $state.go("app.index");
            return;
          }
          _ref = $rootScope.history;
          _results = [];
          for (_i = 0, _len = _ref.length; _i < _len; _i++) {
            element = _ref[_i];
            element = $rootScope.history[$rootScope.history.length - 1];
            if (element.name === "app.profile" || $state.$current.name === "app.profile") {
              $log.info("UserCtrl redirect to app.index");
              $state.go("app.index");
              break;
            } else {
              _results.push(void 0);
            }
          }
          return _results;
        };
        $scope.$on("$stateChangeSuccess", function(event, toState, toParams, fromState, fromParams) {
          $log.info("MenuDirective Site state changed");
          $log.debug({
            message: "MenuDirective Site state changed",
            event: event,
            toState: toState,
            toParams: toParams,
            fromState: fromState,
            fromParams: fromParams
          });
          return $timeout(function() {
            $("#menu").jPushMenu();
            return $log.info("MenuDirective Menu initialized");
          });
        });
        return $scope.toggleMenu = function() {
          $log.info("MenuDirective toogleMenu called");
          $('#menu,body,.cbp-spmenu').removeClass("disabled active cbp-spmenu-open cbp-spmenu-push-toleft cbp-spmenu-push-toright menu-active");
          return true;
        };
      }
    };
  }
]);
