"use strict";
var app;
 
app = angular.module("app.factories");
 
app.factory("CatalogFactory", [
  "$resource", "$rootScope", function($resource, $rootScope) {
    var CatalogFactory;
    CatalogFactory = (function() {
      function CatalogFactory() {}
 
      CatalogFactory.prototype.getDatasets = function() {
        return $resource($rootScope.apiBase + "/v0/datasets/");
      };
 
      CatalogFactory.prototype.getCategories = function() {
        return $resource($rootScope.apiBase + "/v0/categories/");
      };
 
      return CatalogFactory;
 
    })();
    return new CatalogFactory;
  }
]);
