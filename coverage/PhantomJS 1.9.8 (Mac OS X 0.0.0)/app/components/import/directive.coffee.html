"use strict";
var app;
 
app = angular.module("app.directives");
 
app.directive('ngFileDropStart', [
  "$log", function($log) {
    return {
      link: function($scope, el) {
        el.bind("dragover", function(e) {
          e.preventDefault();
          return false;
        });
        return el.bind("dragenter", function(e) {
          $log.info("ngFileDropStart dragenter called");
          document.getElementById("drop-zone").style.display = "block";
          e.preventDefault();
          return false;
        });
      }
    };
  }
]);
 
app.directive('ngFileDropEnd', [
  "$log", function($log) {
    return {
      link: function($scope, el) {
        el.bind("dragleave", function(e) {
          $log.info("ngFileDropEnd dragleave called");
          document.getElementById("drop-zone").style.display = "none";
          e.preventDefault();
          return false;
        });
        el.bind("dragover", function(e) {
          e.preventDefault();
          return false;
        });
        return el.bind("drop", function(e) {
          $log.info("ngFileDropEnd drop called");
          e.preventDefault();
          document.getElementById("drop-zone").style.display = "none";
          $scope.file = e.originalEvent.dataTransfer.files[0];
          $scope.getFile("dragged");
          return false;
        });
      }
    };
  }
]);
 
app.directive('ngFileSelect', [
  "$log", function($log) {
    return {
      link: function($scope, el) {
        return el.bind("change", function(e) {
          $log.info("ngFileSelect change called");
          $scope.file = (e.srcElement || e.target).files[0];
          return $scope.getFile();
        });
      }
    };
  }
]);
