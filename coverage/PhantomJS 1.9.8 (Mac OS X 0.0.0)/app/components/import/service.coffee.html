"use strict";
var app;
 
app = angular.module("app.services");
 
app.service('ImportService', [
  "$q", "$log", function($q, $log) {
    var Reader;
    Reader = (function() {
      function Reader() {
        $log.info("ImportService constructor called");
        this.reader = new FileReader();
        this.deferred = void 0;
        this.progress = 0;
      }
 
      Reader.prototype.readFile = function(file, fileType) {
        $log.info("ImportService readFile called");
        $log.debug({
          message: "ImportService readFile called",
          file: file,
          fileType: fileType
        });
        this.deferred = $q.defer();
        this.progress = 0;
        this.reader.onload = (function(_this) {
          return function() {
            return _this.deferred.resolve(_this.reader.result);
          };
        })(this);
        this.reader.onerror = (function(_this) {
          return function() {
            return _this.deferred.reject(_this.reader.result);
          };
        })(this);
        switch (fileType) {
          case "csv":
            this.reader.readAsText(file);
            break;
          case "zip":
            this.reader.readAsArrayBuffer(file);
        }
        return this.deferred.promise;
      };
 
      return Reader;
 
    })();
    return new Reader;
  }
]);
