"use strict";
var app;
 
app = angular.module("app.services");
 
app.service("ProgressService", [
  "$log", function($log) {
    var Progress;
    Progress = (function() {
      var _message;
 
      _message = null;
 
      function Progress() {
        _message = "";
      }
 
      Progress.prototype.getMessage = function() {
        return _message;
      };
 
      Progress.prototype.setMessage = function(msg) {
        $log.info("Progress setMessage function called");
        $log.debug({
          message: msg
        });
        return _message = msg;
      };
 
      return Progress;
 
    })();
    return new Progress;
  }
]);
