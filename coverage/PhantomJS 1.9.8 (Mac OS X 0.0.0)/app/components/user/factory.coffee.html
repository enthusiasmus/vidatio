"use strict";
var app;
 
app = angular.module("app.factories");
 
app.factory('UserFactory', [
  "$resource", "$rootScope", function($resource, $rootScope) {
    return $resource($rootScope.apiBase + $rootScope.apiVersion + "/users/:id");
  }
]);
 
app.factory('UserAuthFactory', [
  "$resource", "$rootScope", function($resource, $rootScope) {
    return $resource($rootScope.apiBase + $rootScope.apiVersion + "/auth");
  }
]);
 
app.factory('UserUniquenessFactory', [
  "$resource", "$rootScope", function($resource, $rootScope) {
    return $resource($rootScope.apiBase + $rootScope.apiVersion + "/users/check", null, {
      check: {
        method: "GET"
      }
    });
  }
]);
