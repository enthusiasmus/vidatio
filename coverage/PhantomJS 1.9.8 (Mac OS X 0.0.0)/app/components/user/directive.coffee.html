"use strict";
var app;
 
app = angular.module("app.directives");
 
app.directive("validateUniqueness", [
  "$q", "UserService", function($q, UserService) {
    return {
      restrict: "A",
      require: 'ngModel',
      link: function(scope, element, attributes, controller) {
        var property;
        property = attributes.validateUniqueness;
        return controller.$asyncValidators.uniqueness = function(modelValue, viewValue) {
          return $q(function(resolve, reject) {
            if ((viewValue != null) && viewValue.length) {
              return UserService.checkUniqueness(property, viewValue).then(function() {
                return resolve();
              }, function(error) {
                return reject();
              });
            }
          });
        };
      }
    };
  }
]);
