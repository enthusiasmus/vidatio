"use strict";
var app;
 
app = angular.module("app.factories");
 
app.factory("DataFactory", [
  "$resource", "$rootScope", function($resource, $rootScope) {
    return $resource($rootScope.apiBase + $rootScope.apiVersion + "/datasets/:id");
  }
]);
