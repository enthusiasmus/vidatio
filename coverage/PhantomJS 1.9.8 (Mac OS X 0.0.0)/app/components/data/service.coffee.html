"use strict";
var app;
 
app = angular.module("app.services");
 
app.service('DataService', [
  "MapService", "TableService", "ConverterService", "VisualizationService", "$rootScope", "ngToast", "$translate", "$log", "DataFactory", "$location", "$state", function(Map, Table, Converter, Visualization, $rootScope, ngToast, $translate, $log, DataFactory, $location, $state) {
    var Data;
    Data = (function() {
      function Data() {
        $log.info("DataService constructor called");
        this.meta = {
          "fileType": "",
          "fileName": ""
        };
      }
 
      Data.prototype.updateMap = function(row, column, oldData, newData) {
        var columnHeaders, key;
        $log.info("DataService updateMap called");
        $log.debug({
          message: "DataService updateMap called",
          row: row,
          column: column,
          oldData: oldData,
          newData: newData
        });
        columnHeaders = Table.instanceTable.getColHeader();
        key = columnHeaders[column];
        return Map.updateGeoJSONWithSHP(row, column, oldData, newData, key);
      };
 
      Data.prototype.validateInput = function(row, column, oldData, newData) {
        var columnHeaders, key;
        $log.info("DataService validateInput called");
        $log.debug({
          message: "DataService validateInput called",
          row: row,
          column: column,
          oldData: oldData,
          newData: newData
        });
        columnHeaders = Table.instanceTable.getColHeader();
        key = columnHeaders[column];
        return Map.validateGeoJSONUpdateSHP(row, column, oldData, newData, key);
      };
 
      Data.prototype.saveViaAPI = function(dataset, name) {
        $log.info("saveViaAPI called");
        $log.debug({
          dataset: dataset,
          name: name
        });
        return DataFactory.save({
          name: name,
          data: dataset,
          metaData: {
            fileType: this.meta.fileType,
            fileName: this.meta.fileName
          },
          options: {
            diagramType: Visualization.options.diagramType,
            xAxisCurrent: Visualization.options.xAxisCurrent,
            yAxisCurrent: Visualization.options.yAxisCurrent,
            color: Visualization.options.color,
            selectedDiagramName: Visualization.options.selectedDiagramName,
            useColumnHeadersFromDataset: Table.useColumnHeadersFromDataset
          }
        }, function(response) {
          var link;
          $log.info("Dataset successfully saved");
          $log.debug({
            response: response
          });
          $translate('TOAST_MESSAGES.DATASET_SAVED').then(function(translation) {
            return ngToast.create({
              content: translation
            });
          });
          link = $state.href("app.dataset", {
            id: response._id
          }, {
            absolute: true
          });
          $rootScope.link = link;
          return $rootScope.showLink = true;
        }, function(error) {
          $log.error("Dataset couldn't be saved");
          $log.debug({
            error: error
          });
          return $translate('TOAST_MESSAGES.DATASET_NOT_SAVED').then(function(translation) {
            return ngToast.create({
              content: translation,
              className: "danger"
            });
          });
        });
      };
 
      return Data;
 
    })();
    return new Data;
  }
]);
