"use strict";
var app;
 
app = angular.module("app.services");
 
app.service('VisualizationService', [
  "TableService", "ConverterService", "MapService", "$log", "$translate", "ngToast", function(Table, Converter, Map, $log, $translate, ngToast) {
    var Visualization;
    Visualization = (function() {
      var initInlineEditingLabels;
 
      function Visualization() {
        this.options = {
          diagramType: false,
          xAxisCurrent: 0,
          yAxisCurrent: 1,
          color: "#11DDC6",
          selectedDiagramName: null,
          translationKeys: {
            "scatter": "DIAGRAMS.SCATTER_PLOT",
            "map": "DIAGRAMS.MAP",
            "parallel": "DIAGRAMS.PARALLEL_COORDINATES",
            "bar": "DIAGRAMS.BAR_CHART",
            "timeseries": "DIAGRAMS.TIME_SERIES"
          }
        };
      }
 
      Visualization.prototype.recommendDiagram = function() {
        var recommendationResults, trimmedDataset;
        $log.info("VisualizationService recommend called");
        trimmedDataset = vidatio.helper.trimDataset(Table.getDataset());
        recommendationResults = vidatio.recommender.run(trimmedDataset, Table.getHeader(), Table.useColumnHeadersFromDataset);
        Eif (recommendationResults.error != null) {
          $log.error("Visualization Ctrl error at recommend diagram");
          $log.debug({
            error: recommendationResults.error
          });
          return this.options.diagramType = false;
        } else {
          $log.info("VisualizationCtrl recommender results: " + (JSON.stringify(recommendationResults)));
          this.options.diagramType = recommendationResults.type;
          this.options.xAxisCurrent = String(recommendationResults.xColumn);
          this.options.yAxisCurrent = String(recommendationResults.yColumn);
          $translate(this.options.translationKeys[this.options.diagramType]).then((function(_this) {
            return function(translation) {
              return _this.options.selectedDiagramName = translation;
            };
          })(this));
          return Table.setDiagramColumns(recommendationResults.xColumn, recommendationResults.yColumn);
        }
      };
 
      Visualization.prototype.isInputValid = function(x, y, diagramType) {
        var subset, transposedDataset, xColumnType, xSubsetFiltered, yColumnType, ySubsetFiltered;
        $log.info("VisualizationService isInputValid called");
        $log.debug({
          x: x,
          y: y,
          diagramType: diagramType
        });
        transposedDataset = vidatio.helper.transposeDataset(Table.getDataset());
        subset = vidatio.helper.getSubset(transposedDataset);
        xSubsetFiltered = subset[x].filter(function(value) {
          if (value != null) {
            return value;
          }
        });
        ySubsetFiltered = subset[y].filter(function(value) {
          if (value != null) {
            return value;
          }
        });
        xColumnType = vidatio.recommender.getColumnType(xSubsetFiltered);
        yColumnType = vidatio.recommender.getColumnType(ySubsetFiltered);
        return vidatio.helper.isDiagramPossible(xColumnType, yColumnType, diagramType);
      };
 
      Visualization.prototype.create = function(options) {
        var chartData, geoJSON, headers, schema, subset, transposedDataset, trimmedDataset;
        $log.info("VisualizationService create function called");
        $log.debug({
          options: options
        });
        trimmedDataset = vidatio.helper.trimDataset(Table.getDataset());
        headers = Table.getHeader();
        options["headers"] = {
          "x": headers[options.xColumn],
          "y": headers[options.yColumn]
        };
        subset = vidatio.helper.getSubset(trimmedDataset);
        transposedDataset = vidatio.helper.transposeDataset(subset);
        schema = vidatio.recommender.getSchema(transposedDataset);
        if (vidatio.helper.isDiagramPossible(schema[options.xColumn], schema[options.yColumn], options.type)) {
          chartData = trimmedDataset;
        } else {
          $translate('TOAST_MESSAGES.NO_DIAGRAM_POSSIBLE').then(function(translation) {
            return ngToast.create({
              className: "danger",
              content: translation
            });
          });
          chartData = [];
        }
        switch (options.type) {
          case "scatter":
            new vidatio.ScatterPlot(chartData, options);
            break;
          case "map":
            geoJSON = Converter.convertArrays2GeoJSON(chartData, Table.getHeader(), {
              x: options.xColumn,
              y: options.yColumn
            });
            Map.setInstance();
            Map.setGeoJSON(geoJSON);
            break;
          case "parallel":
            new vidatio.ParallelCoordinates(chartData, options);
            break;
          case "bar":
            new vidatio.BarChart(chartData, options);
            break;
          case "timeseries":
            new vidatio.TimeseriesChart(chartData, options);
            break;
          default:
            $log.error("VisualizationCtrl type not set");
            $log.debug({
              type: options.type
            });
        }
        return initInlineEditingLabels();
      };
 
      initInlineEditingLabels = function() {
        $log.info("VisualizationService initInlineEditingLabels called");
        return $("#chart").on("click", "#d3plus_graph_xlabel, #d3plus_graph_ylabel", function(event) {
          var $element, axis, defaultCss, inputPosition, inputSize, minWidth, offsetTop;
          $element = $(this);
          minWidth = 200;
          inputSize = {
            width: $element.width() + 20,
            height: ($element.height() || 14) + 16
          };
          if (inputSize.width < minWidth) {
            inputSize.width = minWidth;
          }
          defaultCss = {
            "min-width": "" + minWidth + "px",
            "width": "" + inputSize.width + "px",
            "height": "" + inputSize.height + "px"
          };
          if ($element.attr("id") === "d3plus_graph_ylabel") {
            inputPosition = {
              "left": "-" + ((inputSize.width / 2) - 6) + "px",
              "top": "50%",
              "margin-top": "-" + inputSize.height + "px",
              "transform": "rotate(-90deg)"
            };
            axis = "Y";
          } else {
            offsetTop = $("svg#d3plus").height() - inputSize.height;
            inputPosition = {
              "left": "50%",
              "top": "" + offsetTop + "px",
              "margin-left": "-" + (inputSize.width / 2) + "px"
            };
            axis = "X";
          }
          return $("<input id='" + ($element.attr("id")) + "_inline_input' type='text' value='" + ($element.text()) + "' />").css(angular.extend(defaultCss, inputPosition)).appendTo("#chart").focus().select().blur(function(event) {
            $element.text($(this).val() || axis);
            return $(this).remove();
          }).keyup(function(event) {
            if (event.keyCode === 13 || event.keyCode === 27) {
              $element.text($(this).val() || axis);
              return $(this).remove();
            }
          });
        });
      };
 
      return Visualization;
 
    })();
    return new Visualization;
  }
]);
