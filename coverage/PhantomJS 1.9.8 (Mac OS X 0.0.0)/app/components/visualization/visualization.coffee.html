"use strict";
window.vidatio.Visualization = (function() {
  function Visualization(dataset, color) {
    this.dataset = dataset;
    this.color = color != null ? color : "#11DDC6";
    vidatio.log.info("Visualization constructor called");
    vidatio.log.debug({
      dataset: this.dataset,
      color: this.color
    });
  }
 
  Visualization.prototype.remove = function() {
    vidatio.log.info("Visualization remove called");
    return $("#chart").empty();
  };
 
  Visualization.prototype.preProcess = function(options) {
    var headers, type, xColumn, yColumn;
    vidatio.log.info("Visualization preprocess called");
    vidatio.log.debug({
      options: options
    });
    type = options.type, xColumn = options.xColumn, yColumn = options.yColumn, headers = options.headers;
    return this.chartData = vidatio.helper.transformToArrayOfObjects(this.dataset, xColumn, yColumn, type, headers, this.color);
  };
 
  return Visualization;
 
})();
