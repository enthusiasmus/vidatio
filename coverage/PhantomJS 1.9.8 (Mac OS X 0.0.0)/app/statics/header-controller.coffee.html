var app;
 
app = angular.module("app.controllers");
 
app.controller("HeaderCtrl", [
  "$scope", "$rootScope", "$timeout", "MapService", "DataService", "$log", "ngToast", "$translate", "TableService", function($scope, $rootScope, $timeout, Map, Data, $log, ngToast, $translate, Table) {
    $scope.header = Data;
    $translate("NEW_VIDATIO").then(function(translation) {
      return $scope.standardTitle = translation;
    });
    if (Table.getDataset().length === 1) {
      $scope.header.editorNotInitialized = true;
    } else {
      $scope.header.editorNotInitialized = false;
    }
    return $scope.saveDataset = function() {
      var dataset;
      $log.info("HeaderCtrl saveDataset called");
      if ($scope.header.name === $scope.standardTitle) {
        $scope.header.name = $scope.header.name + " " + moment().format("HH_mm_ss");
      }
      if (Data.meta.fileType === "shp") {
        dataset = Map.getGeoJSON();
      } else {
        dataset = Table.dataset.slice();
        if (Table.useColumnHeadersFromDataset) {
          dataset.unshift(Table.instanceTable.getColHeader());
        }
      }
      return Data.saveViaAPI(dataset);
    };
  }
]);
