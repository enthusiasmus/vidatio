var app;
 
app = angular.module("app.controllers");
 
app.controller("HeaderCtrl", [
  "$scope", "$rootScope", "$timeout", "DataService", "$log", "ngToast", "$translate", "TableService", function($scope, $rootScope, $timeout, Map, Data, $log, ngToast, $translate, Table) {
    $scope.saveDataset = function() {
      var dataset;
      $log.info("HeaderCtrl saveDataset called");
      if (Data.meta.fileType === "shp") {
        dataset = Map.getGeoJSON();
      } else {
        dataset = Table.dataset.slice();
        if (Table.useColumnHeadersFromDataset) {
          dataset.unshift(Table.instanceTable.getColHeader());
        }
      }
      return Data.saveViaAPI(dataset);
    };
    $scope.hideLink = function() {
      return $rootScope.showLink = false;
    };
    return $scope.copyLink = function() {
      var err, link, range, successful;
      $log.info("HeaderCtrl copyLink called");
      window.getSelection().removeAllRanges();
      link = document.querySelector("#link");
      range = document.createRange();
      range.selectNode(link);
      window.getSelection().addRange(range);
      try {
        successful = document.execCommand("copy");
        $log.debug({
          message: "HeaderCtrl copyLink copy link to clipboard",
          successful: successful
        });
        $translate("TOAST_MESSAGES.LINK_COPIED").then(function(translation) {
          return ngToast.create({
            content: translation
          });
        });
      } catch (_error) {
        err = _error;
        $log.info("Link could not be copied");
        $log.error({
          error: error
        });
      }
      return window.getSelection().removeAllRanges();
    };
  }
]);
