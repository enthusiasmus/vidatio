"use strict";
var app;
 
app = angular.module("app.controllers");
 
app.controller("LoginCtrl", [
  "$scope", "UserService", "$rootScope", "$state", "$log", "$translate", "ngToast", "$stateParams", function($scope, UserService, $rootScope, $state, $log, $translate, ngToast) {
    $scope.logon = function() {
      $log.info("LoginCtrl logon called");
      return UserService.logon($scope.user).then(function(value) {
        var element, _i, _len, _ref;
        $log.info("LoginCtrl successfully logged in");
        $log.debug({
          value: value
        });
        if (!$rootScope.history.length) {
          $log.info("LoginCtrl redirect to app.index");
          $state.go("app.index");
          return;
        }
        _ref = $rootScope.history;
        for (_i = 0, _len = _ref.length; _i < _len; _i++) {
          element = _ref[_i];
          element = $rootScope.history[$rootScope.history.length - 1];
          if (element.name !== "app.login" && element.name !== "app.registration" && element.name !== "") {
            $log.info("UserCtrl redirect to " + element.name);
            if (element.name === "app.dataset") {
              $state.go(element.name, {
                'id': element.params.id
              }, element.params.locale);
              return;
            }
            $state.go(element.name, element.params.locale);
            return;
          }
        }
        return $state.go("app.index");
      }, function(error) {
        $log.info("LoginCtrl error on login");
        return $translate('TOAST_MESSAGES.NOT_AUTHORIZED').then(function(translation) {
          return ngToast.create({
            content: translation,
            className: "danger"
          });
        });
      });
    };
    return $('.input-group').on('focus', '.form-control', function() {
      return $(this).closest('.input-group, .form-group').addClass('focus');
    }).on('blur', '.form-control', function() {
      return $(this).closest('.input-group, .form-group').removeClass('focus');
    });
  }
]);
