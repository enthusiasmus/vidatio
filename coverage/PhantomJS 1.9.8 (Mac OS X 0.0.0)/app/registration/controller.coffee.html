"use strict";
var app;
 
app = angular.module("app.controllers");
 
app.controller("RegistrationCtrl", [
  "$scope", "UserFactory", "$log", "$translate", "ngToast", "$state", function($scope, UserFactory, $log, $translate, ngToast, $state) {
    $log.info("RegistrationCtrl called");
    $scope.register = function() {
      $log.info("RegistrationCtrl register called");
      return UserFactory.save($scope.user, function(response) {
        $log.info("RegistrationCtrl register success called");
        $log.debug({
          response: response
        });
        $translate('TOAST_MESSAGES.REGISTRATION_SUCCESS').then(function(translation) {
          return ngToast.create({
            content: translation
          });
        });
        return $state.go("app.login");
      }, function(error) {
        $log.error("RegistrationCtrl register error called");
        $log.debug({
          error: error
        });
        return $translate('TOAST_MESSAGES.SERVER_ERROR').then(function(translation) {
          return ngToast.create({
            content: translation,
            className: "danger"
          });
        });
      });
    };
    return $('.input-group').on('focus', '.form-control', function() {
      return $(this).closest('.input-group, .form-group').addClass('focus');
    }).on('blur', '.form-control', function() {
      return $(this).closest('.input-group, .form-group').removeClass('focus');
    });
  }
]);
